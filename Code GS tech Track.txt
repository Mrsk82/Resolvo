function validateEmail(email) {
  const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
  return emailRegex.test(email);
}

function validateRequired(value, fieldName) {
  if (!value || (typeof value === 'string' && value.trim() === '')) {
    throw new Error(`${fieldName} is required`);
  }
  return value.trim();
}

function validatePriority(priority) {
  const validPriorities = ['Critical', 'High', 'Medium', 'Low'];
  if (!validPriorities.includes(priority)) {
    throw new Error(`Invalid priority. Must be one of: ${validPriorities.join(', ')}`);
  }
  return priority;
}

function validateStatus(status) {
  const validStatuses = ['Open', 'Acknowledged', 'WIP', 'Testing', 'Resolved', 'Closed', 'Blocked', 'Need Info', 'Reopened'];
  if (!validStatuses.includes(status)) {
    throw new Error(`Invalid status. Must be one of: ${validStatuses.join(', ')}`);
  }
  return status;
}

function sanitizeString(str, maxLength = 1000) {
  if (typeof str !== 'string') return '';
  return str.trim().substring(0, maxLength);
}

function getCurrentUser() {
  try {
    const email = Session.getActiveUser().getEmail();
    if (!email) return { success: false, error: 'Could not detect your Google account. Make sure you are logged in.' };
    
    let ss;
    try {
      ss = getSpreadsheet();
    } catch(e) {
      return { success: false, error: e.message };
    }
    
    const usersSheet = ss.getSheetByName(SHEETS.USERS);
    if (!usersSheet) {
      return { success: false, error: 'USERS sheet not found. Please run initializeSpreadsheet() first from the Apps Script editor.' };
    }
    
    const data = usersSheet.getDataRange().getValues();
    const headers = data[0];
    
    for (let i = 1; i < data.length; i++) {
      const row = data[i];
      const userEmail = row[headers.indexOf('Email')];
      if (userEmail === email) {
        return {
          success: true,
          user: {
            id: row[headers.indexOf('UserID')],
            email: row[headers.indexOf('Email')],
            name: row[headers.indexOf('Name')],
            team: row[headers.indexOf('Team')],
            role: row[headers.indexOf('Role')],
            skill: row[headers.indexOf('Skill')],
            slackId: row[headers.indexOf('SlackID')],
            maxTickets: row[headers.indexOf('MaxTickets')],
            active: row[headers.indexOf('Active')]
          }
        };
      }
    }
    return { success: false, error: `Your email (${email}) is not registered. Ask an Admin to add you in User Management.` };
  } catch (err) {
    return { success: false, error: err.message };
  }
}

function getUsers(activeOnly) {

    const sheet = ss.getSheetByName(SHEETS.USERS);

    const users = [];
    
    for (let i = 1; i < data.length; i++) {
      const row = data[i];
      const active = row[headers.indexOf('Active')];
      if (activeOnly && active !== true && active !== 'TRUE' && active !== 'true') continue;
      
      users.push({
        id: row[headers.indexOf('UserID')],
        email: row[headers.indexOf('Email')],
        name: row[headers.indexOf('Name')],
        team: row[headers.indexOf('Team')],
        role: row[headers.indexOf('Role')],
        skill: row[headers.indexOf('Skill')],
        slackId: row[headers.indexOf('SlackID')],
        maxTickets: row[headers.indexOf('MaxTickets')],
        active: row[headers.indexOf('Active')],
        createdDate: row[headers.indexOf('CreatedDate')]
      });
    }
    return { success: true, users };
  } catch (err) {
    return { success: false, error: err.message };
  }
}

function updateUser(userId, userData) {
  try {
    const ss = getSpreadsheet();
    const sheet = ss.getSheetByName(SHEETS.USERS);
    const data = sheet.getDataRange().getValues();
    const headers = data[0];

    for (let i = 1; i < data.length; i++) {
      if (data[i][headers.indexOf('UserID')] === userId) {
        const row = i + 1;
        // Read the full current row, apply only the provided fields, then write in one call
        const currentRow = data[i].slice(); // copy
        const fieldMap = {
          name:       'Name',
          team:       'Team',
          role:       'Role',
          skill:      'Skill',
          slackId:    'SlackID',
          maxTickets: 'MaxTickets',
          active:     'Active'
        };
        Object.keys(fieldMap).forEach(function(field) {
          if (userData[field] !== undefined) {
            currentRow[headers.indexOf(fieldMap[field])] = userData[field];
          }
        });
        // Single batch write for the entire row
        sheet.getRange(row, 1, 1, currentRow.length).setValues([currentRow]);
        return { success: true };
      }
    }
    return { success: false, error: 'User not found' };
  } catch (err) {
    return { success: false, error: err.message };
  }
}

function createIssue(issueData) {
  try {
    const ss = getSpreadsheet();
    const sheet = ss.getSheetByName(SHEETS.ISSUES);
    const currentUser = getCurrentUser();
    if (!currentUser.success) return currentUser;
    
    const issueId = generateIssueId();
    const now = new Date();
    const slaHours = getSLAHours(issueData.priority);
    
    // Check for duplicates if enabled
    if (getSetting('DUPLICATE_DETECTION_ENABLED') === 'true') {
      const dupes = checkDuplicates(issueData.title);
      if (dupes.length > 0) {
        return { success: false, duplicates: dupes, message: 'Possible duplicate issues found' };
      }
    }
    
    let assignedTo = issueData.assignedTo || '';
    
    // Auto-assign if enabled
    if (getSetting('AUTO_ASSIGN_ENABLED') === 'true' && !assignedTo) {
      const autoResult = autoAssignDeveloper(issueData.priority, issueData.module);
      if (autoResult.success) assignedTo = autoResult.email;
    }
    
    sheet.appendRow([
      issueId,
      issueData.title,
      issueData.description,
      issueData.module || '',
      issueData.priority,
      'Open',
      issueData.environment || '',
      currentUser.user.email,
      assignedTo,
      now.toISOString(),
      '', // StartedDate
      '', // ResolvedDate
      '', // ClosedDate
      issueData.impact || '',
      slaHours,
      issueData.attachmentUrl || ''
    ]);
    
    logActivity(issueId, 'Issue Created', currentUser.user.email);
    
    if (assignedTo) {
      logActivity(issueId, 'Assigned to ' + assignedTo, currentUser.user.email);
    }

    // Send notification email and link the Gmail thread for tracking
    try {
      sendIssueCreationNotification(issueId, issueData, assignedTo, currentUser.user);
    } catch(notifErr) {
      console.error('Notification error:', notifErr.message);
    }
    
    // Slack alert for critical issues
    if (issueData.priority === 'Critical') {
      const critMsg = '🚨 CRITICAL ISSUE: ' + issueId + ' — ' + issueData.title + ' (by ' + currentUser.user.name + ', SLA: ' + slaHours + 'h)';
      sendSlackAlert(critMsg);
      sendWebhookAlert(critMsg, 'Critical');
    }
    
    return { success: true, issueId };
  } catch (err) {
    return { success: false, error: err.message };
  }
}

function getIssues(filters) {
  try {
    const ss = getSpreadsheet();
    const sheet = ss.getSheetByName(SHEETS.ISSUES);
    const data = sheet.getDataRange().getValues();
    const headers = data[0];
    const issues = [];
    const now = new Date();
    
    for (let i = 1; i < data.length; i++) {
      const row = data[i];
      if (!row[0]) continue;
      
      const issue = {
        id: row[headers.indexOf('IssueID')],
        title: row[headers.indexOf('Title')],
        description: row[headers.indexOf('Description')],
        module: row[headers.indexOf('Module')],
        priority: row[headers.indexOf('Priority')],
        status: row[headers.indexOf('Status')],
        environment: row[headers.indexOf('Environment')],
        raisedBy: row[headers.indexOf('RaisedBy')],
        assignedTo: row[headers.indexOf('AssignedTo')],
        createdDate: row[headers.indexOf('CreatedDate')],
        startedDate: row[headers.indexOf('StartedDate')],
        resolvedDate: row[headers.indexOf('ResolvedDate')],
        closedDate: row[headers.indexOf('ClosedDate')],
        impact: row[headers.indexOf('Impact')],
        slaHours: row[headers.indexOf('SLAHours')],
        attachmentUrl: row[headers.indexOf('AttachmentURL')],
        sprintId: row[headers.indexOf('SprintID')] || ''
      };
      
      // Compute SLA data
      const createdDate = new Date(issue.createdDate);
      const slaDeadline = new Date(createdDate.getTime() + issue.slaHours * 60 * 60 * 1000);
      const hoursRemaining = (slaDeadline - now) / (1000 * 60 * 60);
      
      issue.slaDeadline = slaDeadline.toISOString();
      issue.slaBreached = (now > slaDeadline) && !['Resolved', 'Release Required'].includes(issue.status);
      issue.slaHoursRemaining = Math.round(hoursRemaining * 10) / 10;
      issue.slaRisk = hoursRemaining > 0 && hoursRemaining < (issue.slaHours * 0.2);
      
      // Apply filters
      if (filters) {
        if (filters.status && filters.status !== 'all' && issue.status !== filters.status) continue;
        if (filters.priority && filters.priority !== 'all' && issue.priority !== filters.priority) continue;
        if (filters.module && filters.module !== 'all' && issue.module !== filters.module) continue;
        if (filters.assignedTo && filters.assignedTo !== 'all' && issue.assignedTo !== filters.assignedTo) continue;
        if (filters.raisedBy && issue.raisedBy !== filters.raisedBy) continue;
        if (filters.dateFrom && new Date(issue.createdDate) < new Date(filters.dateFrom)) continue;
        if (filters.dateTo && new Date(issue.createdDate) > new Date(filters.dateTo)) continue;
      }
      
      issues.push(issue);
    }
    
    return { success: true, issues: issues.reverse() };
  } catch (err) {
    return { success: false, error: err.message };
  }
}

function updateIssueStatus(issueId, newStatus, comment) {
  try {
    const ss = getSpreadsheet();
    const sheet = ss.getSheetByName(SHEETS.ISSUES);
    const data = sheet.getDataRange().getValues();
    const headers = data[0];
    const currentUser = getCurrentUser();
    if (!currentUser.success) return currentUser;
    
    const now = new Date().toISOString();
    
    for (let i = 1; i < data.length; i++) {
      if (data[i][headers.indexOf('IssueID')] === issueId) {
        const row = i + 1;
        sheet.getRange(row, headers.indexOf('Status') + 1).setValue(newStatus);
        
        if (newStatus === 'WIP' && !data[i][headers.indexOf('StartedDate')]) {
          sheet.getRange(row, headers.indexOf('StartedDate') + 1).setValue(now);
        }
        if (newStatus === 'Resolved') {
          sheet.getRange(row, headers.indexOf('ResolvedDate') + 1).setValue(now);
        }
        if (newStatus === 'Release Required') {
          sheet.getRange(row, headers.indexOf('ResolvedDate') + 1).setValue(now);
        }
        
        logActivity(issueId, `Status changed to ${newStatus}`, currentUser.user.email);
        
        if (comment) addComment(issueId, comment);
        
        const title = data[i][headers.indexOf('Title')];
        const assignedTo = data[i][headers.indexOf('AssignedTo')];
        const raisedBy = data[i][headers.indexOf('RaisedBy')];
        
        sendStatusUpdateEmail(issueId, title, newStatus, assignedTo, raisedBy);
        
        if (newStatus === 'Resolved') {
          sendSlackAlert(`✅ *ISSUE RESOLVED*\n*ID:* ${issueId}\n*Title:* ${title}\n*By:* ${currentUser.user.name}`);
        }
        
        return { success: true };
      }
    }
    return { success: false, error: 'Issue not found' };
  } catch (err) {
    return { success: false, error: err.message };
  }
}

function checkDuplicates(title) {
  const result = getIssues(null);
  if (!result.success) return [];
  const words = title.toLowerCase().split(' ').filter(w => w.length > 3);
  return result.issues.filter(issue => {
    if (['Release Required', 'Resolved'].includes(issue.status)) return false;
    const issueWords = issue.title.toLowerCase().split(' ').filter(w => w.length > 3);
    const matches = words.filter(w => issueWords.includes(w));
    return matches.length >= Math.min(3, words.length * 0.5);
  }).slice(0, 5);
}

function autoAssignDeveloper(priority, module) {
  try {
    const devsResult = getDevelopers();
    if (!devsResult.success || devsResult.developers.length === 0) return { success: false };
    
    const issuesResult = getIssues({ status: 'Open' });
    const activeIssues = issuesResult.success ? issuesResult.issues : [];
    
    let bestDev = null;
    let lowestLoad = Infinity;
    
    for (const dev of devsResult.developers) {
      const devIssues = activeIssues.filter(i => i.assignedTo === dev.email && 
        !['Resolved', 'Closed'].includes(i.status));
      const load = devIssues.length;
      const maxTickets = parseInt(dev.maxTickets) || 10;
      
      if (load < maxTickets && load < lowestLoad) {
        lowestLoad = load;
        bestDev = dev;
      }
    }
    
    if (bestDev) return { success: true, email: bestDev.email };
    return { success: false };
  } catch (err) {
    return { success: false };
  }
}

function getSettings() {
  try {
    // Check cache first (2-minute TTL)
    const cache = CacheService.getScriptCache();
    const cached = cache.get('SYSTEM_SETTINGS');
    if (cached) {
      return { success: true, settings: JSON.parse(cached) };
    }

    const ss = getSpreadsheet();
    const sheet = ss.getSheetByName(SHEETS.SYSTEM_SETTINGS);
    const data = sheet.getDataRange().getValues();
    const headers = data[0];
    const settings = {};

    for (let i = 1; i < data.length; i++) {
      settings[data[i][headers.indexOf('SettingKey')]] = data[i][headers.indexOf('SettingValue')];
    }

    // Store in cache for 120 seconds
    try { cache.put('SYSTEM_SETTINGS', JSON.stringify(settings), 120); } catch(ce) {}

    return { success: true, settings };
  } catch (err) {
    return { success: false, error: err.message };
  }
}

function getSetting(key) {
  const result = getSettings();
  if (!result.success) return 'false';
  return result.settings[key] || 'false';
}

function updateSetting(key, value) {
  try {
    const ss = getSpreadsheet();
    const sheet = ss.getSheetByName(SHEETS.SYSTEM_SETTINGS);
    const data = sheet.getDataRange().getValues();
    const headers = data[0];

    for (let i = 1; i < data.length; i++) {
      if (data[i][headers.indexOf('SettingKey')] === key) {
        sheet.getRange(i + 1, headers.indexOf('SettingValue') + 1).setValue(value);
        // Invalidate settings cache on every write
        try { CacheService.getScriptCache().remove('SYSTEM_SETTINGS'); } catch(ce) {}
        return { success: true };
      }
    }
    sheet.appendRow([key, value]);
    // Invalidate settings cache on every write
    try { CacheService.getScriptCache().remove('SYSTEM_SETTINGS'); } catch(ce) {}
    return { success: true };
  } catch (err) {
    return { success: false, error: err.message };
  }
}

function getDashboardStats() {
  try {
    const issuesResult = getIssues(null);
    if (!issuesResult.success) return issuesResult;
    
    const issues = issuesResult.issues;
    const stats = {
      total: issues.length,
      open: issues.filter(i => i.status === 'Open').length,
      inProgress: issues.filter(i => i.status === 'WIP').length,
      critical: issues.filter(i => i.priority === 'Critical' && !['Resolved','Release Required'].includes(i.status)).length,
      slaBreached: issues.filter(i => i.slaBreached).length,
      resolved: issues.filter(i => i.status === 'Resolved').length,
      closed: issues.filter(i => i.status === 'Release Required').length
    };
    
    // Issues by module
    const moduleCount = {};
    issues.forEach(i => {
      if (i.module) moduleCount[i.module] = (moduleCount[i.module] || 0) + 1;
    });
    
    // Issues by priority
    const priorityCount = { Critical: 0, High: 0, Medium: 0, Low: 0 };
    issues.forEach(i => {
      if (priorityCount[i.priority] !== undefined) priorityCount[i.priority]++;
    });
    
    // Developer workload
    const devWorkload = {};
    issues.filter(i => !['Resolved','Release Required'].includes(i.status)).forEach(i => {
      if (i.assignedTo) devWorkload[i.assignedTo] = (devWorkload[i.assignedTo] || 0) + 1;
    });
    
    // Resolution trend (last 7 days)
    const trend = [];
    for (let d = 6; d >= 0; d--) {
      const date = new Date();
      date.setDate(date.getDate() - d);
      const dateStr = date.toISOString().split('T')[0];
      const count = issues.filter(i => i.resolvedDate && i.resolvedDate.startsWith(dateStr)).length;
      trend.push({ date: dateStr, count });
    }
    
    // Severity breakdown
    const severityTotal = priorityCount;
    const severityInProgress = issues.filter(i => i.status === 'WIP').reduce((acc, i) => {
      acc[i.priority] = (acc[i.priority] || 0) + 1;
      return acc;
    }, {Critical:0, High:0, Medium:0, Low:0});
    
    // Dev workload severity total
    const devWorkloadSeverityTotal = issues.filter(i => !['Resolved','Release Required'].includes(i.status)).reduce((acc, i) => {
      if (i.assignedTo) {
        if (!acc[i.assignedTo]) acc[i.assignedTo] = {Critical:0, High:0, Medium:0, Low:0};
        acc[i.assignedTo][i.priority]++;
      }
      return acc;
    }, {});
    
    // Dev workload severity WIP
    const devWorkloadSeverityInProgress = issues.filter(i => i.status === 'WIP').reduce((acc, i) => {
      if (i.assignedTo) {
        if (!acc[i.assignedTo]) acc[i.assignedTo] = {Critical:0, High:0, Medium:0, Low:0};
        acc[i.assignedTo][i.priority]++;
      }
      return acc;
    }, {});
    
    // Aging
    const aging = issues.filter(i => i.status === 'WIP').map(i => {
      const days = Math.floor((new Date() - new Date(i.createdDate)) / (1000 * 60 * 60 * 24));
      if (days <= 2) return '0–2 days';
      if (days <= 5) return '3–5 days';
      if (days <= 10) return '6–10 days';
      return '10+ days';
    }).reduce((acc, bucket) => {
      acc[bucket] = (acc[bucket] || 0) + 1;
      return acc;
    }, {'0–2 days':0, '3–5 days':0, '6–10 days':0, '10+ days':0});
    
    return { success: true, stats, moduleCount, priorityCount, devWorkload, trend, severityTotal, severityInProgress, devWorkloadSeverityTotal, devWorkloadSeverityInProgress, aging };
  } catch (err) {
    return { success: false, error: err.message };
  }
}

function getDevPerformance() {
  try {
    const issuesResult = getIssues(null);
    if (!issuesResult.success) return issuesResult;
    const issues = issuesResult.issues;
    const devMap = {};
    
    issues.forEach(issue => {
      if (!issue.assignedTo) return;
      const dev = issue.assignedTo;
      if (!devMap[dev]) devMap[dev] = { email: dev, resolved: 0, totalResolutionHours: 0, reopened: 0 };
      
      if (['Resolved', 'Release Required'].includes(issue.status) && issue.resolvedDate && issue.createdDate) {
        const hours = (new Date(issue.resolvedDate) - new Date(issue.createdDate)) / (1000 * 60 * 60);
        devMap[dev].resolved++;
        devMap[dev].totalResolutionHours += hours;
      }
      if (issue.status === 'Reopened') devMap[dev].reopened++;
    });
    
    const performance = Object.values(devMap).map(dev => ({
      ...dev,
      avgResolutionHours: dev.resolved > 0 ? Math.round(dev.totalResolutionHours / dev.resolved) : 0
    }));
    
    return { success: true, performance };
  } catch (err) {
    return { success: false, error: err.message };
  }
}

function uploadFile(fileName, mimeType, base64Data) {
  try {
    const folderId = DRIVE_FOLDER_ID;
    let folder;
    
    if (folderId) {
      folder = DriveApp.getFolderById(folderId);
    } else {
      const folders = DriveApp.getFoldersByName('TechTrack Attachments');
      folder = folders.hasNext() ? folders.next() : DriveApp.createFolder('TechTrack Attachments');
    }
    
    const blob = Utilities.newBlob(Utilities.base64Decode(base64Data), mimeType, fileName);
    const file = folder.createFile(blob);
    file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
    
    return { success: true, url: file.getUrl(), fileId: file.getId() };
  } catch (err) {
    return { success: false, error: err.message };
  }
}

function checkSLABreaches() {
  if (getSetting('SLA_RISK_ALERTS_ENABLED') !== 'true') return;
  
  const result = getIssues(null);
  if (!result.success) return;
  
  result.issues.forEach(issue => {

      sendSlackAlert(`⚠️ *SLA BREACHED*\n*ID:* ${issue.id}\n*Title:* ${issue.title}\n*Priority:* ${issue.priority}`);
    }
  });
}

function getSpreadsheet() {
  if (!SPREADSHEET_ID || SPREADSHEET_ID === 'PASTE_YOUR_SPREADSHEET_ID_HERE') {
    throw new Error('SPREADSHEET_ID not set in Code.gs. Please paste your Google Sheets ID at the top of Code.gs and redeploy.');
  }
  return SpreadsheetApp.openById(SPREADSHEET_ID);
}

function generateIssueId() {
  const ss = getSpreadsheet();
  const sheet = ss.getSheetByName(SHEETS.ISSUES);
  const lastRow = sheet.getLastRow();
  const num = lastRow; // Row 1 is header
  return 'TECH-' + String(num).padStart(4, '0');
}

function generateUserId() {
  return 'USR-' + Utilities.getUuid().substring(0, 8).toUpperCase();
}

// GEMINI HELPER — backend AI (key stored in Script Properties)
// ============================================================

function callGemini(prompt) {
  try {
    const key = PropertiesService.getScriptProperties().getProperty('GEMINI_API_KEY');
    if (!key) return { success: false, error: 'Gemini API key not set in Script Properties.' };

    const url = 'https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash:generateContent?key=' + key;
    const payload = JSON.stringify({
      contents: [{ parts: [{ text: prompt }] }],
      generationConfig: { temperature: 0.2, maxOutputTokens: 1500, responseMimeType: 'application/json' }
    });

    const response = UrlFetchApp.fetch(url, {
      method: 'post',
      contentType: 'application/json',
      payload: payload,
      muteHttpExceptions: true
    });

    const code = response.getResponseCode();
    const text = response.getContentText();

    if (code !== 200) return { success: false, error: 'Gemini HTTP ' + code + ': ' + text.substring(0, 200) };

    const data = JSON.parse(text);
    const raw = data.candidates && data.candidates[0] && data.candidates[0].content &&
                data.candidates[0].content.parts && data.candidates[0].content.parts[0]
                ? data.candidates[0].content.parts[0].text : '{}';

    const clean = raw.replace(/```json|```/g, '').trim();
    return { success: true, result: JSON.parse(clean) };
  } catch (err) {
    return { success: false, error: err.message };
  }
}

// ============================================================
// BACKEND AI QUERY (replaces browser fetch to Gemini)
// ============================================================

function processAIQuery(query) {
  try {
    const currentUser = getCurrentUser();
    if (!currentUser.success) return { success: false, error: 'Not authenticated' };

    const issuesResult = getIssues(null);
    const issues = issuesResult.success ? issuesResult.issues.slice(0, 50) : [];

    const issuesSummary = issues.map(function(i) {
      return (i.id||'') + '|' + (i.title||'') + '|' + (i.priority||'') + '|' + (i.status||'') + '|' + (i.module||'') + '|' + (i.assignedTo||'');
    }).join('\n');

    const prompt = 'You are an AI assistant for TechTrack engineering issue tracker.\n' +
      'User: ' + currentUser.user.name + ' (' + currentUser.user.role + ')\n' +
      'Query: ' + query + '\n\n' +
      'Return ONLY raw JSON (no markdown):\n' +
      '{"intent":"create|filter|summary|unknown","confidence":0.9,"message":"what you did",' +
      '"filters":[],"issues":[{"id":"","title":"","priority":"","status":"","module":""}],' +
      '"create":{"title":"","description":"","priority":"Medium","module":"","environment":"","impact":""},' +
      '"summary":{"open":0,"critical":0,"breached":0,"resolved":0,"total":0}}\n\n' +
      'Issue data:\n' + issuesSummary + '\n\n' +
      'Rules: create/new/add=create intent; show/find/list=filter; how many/count/summary=summary.\n' +
      'Return matching issues from data for filter. Return only the JSON.';

    const gemResult = callGemini(prompt);
    if (!gemResult.success) {
      return { success: false, error: gemResult.error };
    }

    // Log the query
    try {
      const ss = getSpreadsheet();
      const qSheet = ss.getSheetByName(EMAIL_SHEETS.AI_QUERIES);
      if (qSheet) {
        const qId = 'AQ-' + Utilities.getUuid().substring(0, 8).toUpperCase();
        qSheet.appendRow([qId, currentUser.user.email, query, gemResult.result.intent || 'unknown',
          JSON.stringify(gemResult.result), new Date().toISOString()]);
      }
    } catch(logErr) {}

    return { success: true, result: gemResult.result };
  } catch (err) {
    return { success: false, error: err.message };
  }
}

function getAIQueryHistory(limit) {
  try {
    const currentUser = getCurrentUser();
    if (!currentUser.success) return { success: false, error: 'Not authenticated' };

    const ss = getSpreadsheet();
    const sheet = ss.getSheetByName(EMAIL_SHEETS.AI_QUERIES);
    if (!sheet || sheet.getLastRow() < 2) return { success: true, queries: [] };

    const data = sheet.getDataRange().getValues();
    const headers = data[0];
    const queries = [];
    const isAdmin = currentUser.user.role === 'Admin';

    for (let i = data.length - 1; i >= 1; i--) {
      const row = data[i];
      const userEmail = row[headers.indexOf('UserEmail')];
      if (!isAdmin && userEmail !== currentUser.user.email) continue;
      queries.push({
        id:        row[headers.indexOf('QueryID')],
        userEmail: userEmail,
        query:     row[headers.indexOf('Query')],
        intent:    row[headers.indexOf('Intent')],
        timestamp: row[headers.indexOf('Timestamp')]
      });
      if (queries.length >= (limit || 50)) break;
    }
    return { success: true, queries };
  } catch (err) {
    return { success: false, error: err.message };
  }
}

function sendIssueCreationNotification(issueId, issueData, assignedTo, raisedByUser) {

    if (assignedTo) recipients.push(assignedTo);
    if (raisedByUser && raisedByUser.email) recipients.push(raisedByUser.email);

    // Get all admins and leads
    const usersResult = getUsers(true);
    if (usersResult.success) {
      usersResult.users.forEach(function(u) {
        if (u.role === 'Admin' && recipients.indexOf(u.email) === -1) recipients.push(u.email);
      });
    }

    if (recipients.length === 0) return;

    const priorityColors = { Critical:'#ff4757', High:'#f5a623', Medium:'#74b9ff', Low:'#2ed573' };
    const pColor = priorityColors[issueData.priority] || '#f5a623';

    const htmlBody = '<div style="font-family:sans-serif;max-width:600px;background:#13141d;border-radius:12px;overflow:hidden;">' +
      '<div style="height:3px;background:' + pColor + ';"></div>' +
      '<div style="padding:28px 32px;">' +
      '<p style="margin:0 0 4px;font-size:11px;font-family:monospace;color:#5c6bc0;letter-spacing:0.1em;text-transform:uppercase;">New Issue Created</p>' +
      '<h2 style="margin:0 0 16px;color:#e8eaf6;font-size:20px;">' + issueId + ': ' + issueData.title + '</h2>' +
      '<table style="width:100%;border-collapse:collapse;">' +
      '<tr><td style="padding:6px 0;color:#5c6bc0;font-size:12px;font-family:monospace;width:100px;">Priority</td>' +
      '<td style="padding:6px 0;"><span style="background:' + pColor + '22;color:' + pColor + ';padding:2px 10px;border-radius:999px;font-size:12px;font-weight:700;">' + issueData.priority + '</span></td></tr>' +
      '<tr><td style="padding:6px 0;color:#5c6bc0;font-size:12px;font-family:monospace;">Module</td>' +
      '<td style="padding:6px 0;color:#9fa8da;font-size:13px;">' + (issueData.module||'—') + '</td></tr>' +
      '<tr><td style="padding:6px 0;color:#5c6bc0;font-size:12px;font-family:monospace;">Reporter</td>' +
      '<td style="padding:6px 0;color:#9fa8da;font-size:13px;">' + (raisedByUser ? raisedByUser.name : '') + '</td></tr>' +
      '<tr><td style="padding:6px 0;color:#5c6bc0;font-size:12px;font-family:monospace;">Assigned</td>' +
      '<td style="padding:6px 0;color:#9fa8da;font-size:13px;">' + (assignedTo||'Unassigned') + '</td></tr>' +
      '</table>' +
      '<p style="margin:16px 0 0;font-size:13px;color:#9fa8da;line-height:1.6;">' + (issueData.description||'').substring(0,300) + '</p>' +
      '<div style="margin-top:20px;">' +
      '<a href="' + appUrl + '" style="display:inline-block;padding:10px 24px;background:#f5a623;color:#0d0e14;border-radius:8px;text-decoration:none;font-size:13px;font-weight:700;">View Issue &rarr;</a>' +
      '</div>' +
      '<p style="margin:16px 0 0;font-size:11px;color:#3d4574;">Reply to this email to add updates — they will be captured in TechTrack automatically.</p>' +
      '</div></div>';

    const uniqueRecipients = recipients.filter(function(v, i, a) { return a.indexOf(v) === i; });
    const toEmail = uniqueRecipients.join(',');

    GmailApp.sendEmail(toEmail, subject, 'New issue created: ' + issueId + ' - ' + issueData.title, {
      htmlBody: htmlBody,
      name: 'TechTrack Platform'
    });

    // Find the thread we just sent and link it to the issue
    Utilities.sleep(3000); // wait for Gmail to index it
    try {
      const threads = GmailApp.search('subject:"[' + issueId + ']" from:me', 0, 5);
      if (threads.length > 0) {
        linkThreadToIssue(issueId, threads[0].getId(), subject, uniqueRecipients);
      }
    } catch(linkErr) {
      console.error('Thread link error:', linkErr.message);
    }

  } catch (err) {
    console.error('Issue notification error:', err.message);
  }
}

function linkThreadToIssue(issueId, gmailThreadId, subject, participants) {
  try {
    const ss = getSpreadsheet();
    const sheet = ss.getSheetByName(EMAIL_SHEETS.THREADS);
    if (!sheet) return;

    // Check if already linked
    const data = sheet.getDataRange().getValues();
    for (let i = 1; i < data.length; i++) {
      if (data[i][0] === issueId && data[i][1] === gmailThreadId) return; // already exists
    }

    sheet.appendRow([
      issueId,
      gmailThreadId,
      subject || '',
      new Date().toISOString(),
      JSON.stringify(participants || []),
      0 // messagesProcessed
    ]);
  } catch (err) {
    console.error('linkThreadToIssue error:', err.message);
  }
}

function getLinkedThreads(issueId) {
  try {
    const ss = getSpreadsheet();
    const sheet = ss.getSheetByName(EMAIL_SHEETS.THREADS);
    if (!sheet || sheet.getLastRow() < 2) return { success: true, threads: [] };

    const data = sheet.getDataRange().getValues();
    const headers = data[0];
    const threads = [];

    for (let i = 1; i < data.length; i++) {
      if (!issueId || data[i][headers.indexOf('IssueID')] === issueId) {
        threads.push({
          issueId:           data[i][headers.indexOf('IssueID')],
          threadId:          data[i][headers.indexOf('GmailThreadID')],
          subject:           data[i][headers.indexOf('ThreadSubject')],
          lastScanned:       data[i][headers.indexOf('LastScanned')],
          participants:      safeJsonParse(data[i][headers.indexOf('ParticipantsJSON')], []),
          messagesProcessed: data[i][headers.indexOf('MessagesProcessed')] || 0
        });
      }
    }
    return { success: true, threads };
  } catch (err) {
    return { success: false, error: err.message };
  }
}

// WATCH LINKED THREADS — Time Trigger (every 5 min)
// ============================================================

function watchLinkedThreads() {
  try {
    const ss = getSpreadsheet();
    const threadsSheet = ss.getSheetByName(EMAIL_SHEETS.THREADS);
    if (!threadsSheet || threadsSheet.getLastRow() < 2) return;

    const data = threadsSheet.getDataRange().getValues();
    const headers = data[0];

    for (let i = 1; i < data.length; i++) {
      const issueId   = data[i][headers.indexOf('IssueID')];
      const threadId  = data[i][headers.indexOf('GmailThreadID')];
      const lastScan  = data[i][headers.indexOf('LastScanned')];
      const processed = data[i][headers.indexOf('MessagesProcessed')] || 0;

      // Skip closed/resolved issues
      const issueResult = getIssueById(issueId);
      if (issueResult.success && ['Closed','Resolved'].includes(issueResult.issue.status)) continue;

      try {
        const thread = GmailApp.getThreadById(threadId);
        if (!thread) continue;

        const messages = thread.getMessages();
        const newMessages = messages.slice(processed); // only unprocessed

        if (newMessages.length === 0) continue;

        processNewReplies(issueId, threadId, newMessages, i + 1, threadsSheet, headers);

      } catch (threadErr) {
        console.error('Thread error for ' + issueId + ':', threadErr.message);
      }
    }
  } catch (err) {
    console.error('watchLinkedThreads error:', err.message);
  }
}

// ============================================================
// PROCESS NEW EMAIL REPLIES
// ============================================================

function processNewReplies(issueId, threadId, messages, sheetRow, threadsSheet, headers) {
  try {
    const ss = getSpreadsheet();
    const messagesSheet = ss.getSheetByName(EMAIL_SHEETS.MESSAGES);

    const processedMessages = [];
    const newParticipants = [];

    messages.forEach(function(msg) {
      const sender    = msg.getFrom();
      const senderEmail = extractEmail(sender);
      const senderName  = extractName(sender);
      const timestamp   = msg.getDate().toISOString();
      const rawBody     = msg.getPlainBody();
      const cleanedBody = cleanEmailBody(rawBody);

      if (!cleanedBody || cleanedBody.length < 5) return; // skip empty/quoted-only

      const msgId = 'MSG-' + Utilities.getUuid().substring(0, 8).toUpperCase();

      messagesSheet.appendRow([
        msgId, issueId, threadId, senderEmail, senderName,
        timestamp, cleanedBody, '', // AITags filled later
        msg.getId()
      ]);

      // Add as a comment on the issue
      const commentText = '[Email from ' + (senderName || senderEmail) + ']: ' + cleanedBody.substring(0, 1000);
      const ss2 = getSpreadsheet();
      const commentsSheet = ss2.getSheetByName(SHEETS.COMMENTS);
      const commentId = 'CMT-' + Utilities.getUuid().substring(0, 8).toUpperCase();
      commentsSheet.appendRow([commentId, issueId, senderEmail, commentText, timestamp, 'email']);

      logActivity(issueId, 'Email reply captured from ' + (senderName || senderEmail), 'system');

      processedMessages.push({ sender: senderEmail, name: senderName, body: cleanedBody, timestamp });
      if (newParticipants.indexOf(senderEmail) === -1) newParticipants.push(senderEmail);
    });

    if (processedMessages.length === 0) return;

    // Update thread row: LastScanned + MessagesProcessed count
    const newTotal = (parseInt(threadsSheet.getRange(sheetRow, headers.indexOf('MessagesProcessed') + 1).getValue()) || 0) + processedMessages.length;
    threadsSheet.getRange(sheetRow, headers.indexOf('LastScanned') + 1).setValue(new Date().toISOString());
    threadsSheet.getRange(sheetRow, headers.indexOf('MessagesProcessed') + 1).setValue(newTotal);

    // Update participants on thread row
    const existingP = safeJsonParse(threadsSheet.getRange(sheetRow, headers.indexOf('ParticipantsJSON') + 1).getValue(), []);
    newParticipants.forEach(function(p) {
      if (existingP.indexOf(p) === -1) existingP.push(p);
    });
    threadsSheet.getRange(sheetRow, headers.indexOf('ParticipantsJSON') + 1).setValue(JSON.stringify(existingP));

    // Run AI extraction on the new replies
    extractEmailIntelligence(issueId, processedMessages);

  } catch (err) {
    console.error('processNewReplies error:', err.message);
  }
}

// ============================================================
// AI EXTRACTION FROM EMAIL THREAD
// ============================================================

function extractEmailIntelligence(issueId, newMessages) {
  try {
    if (!newMessages || newMessages.length === 0) return;

    // Get full existing thread context from EMAIL_MESSAGES
    const ss = getSpreadsheet();
    const msgSheet = ss.getSheetByName(EMAIL_SHEETS.MESSAGES);
    const allData  = msgSheet.getDataRange().getValues();
    const allHeaders = allData[0];
    const threadContext = [];

    for (let i = 1; i < allData.length; i++) {
      if (allData[i][allHeaders.indexOf('IssueID')] === issueId) {
        threadContext.push(
          (allData[i][allHeaders.indexOf('SenderName')] || allData[i][allHeaders.indexOf('Sender')]) +
          ' [' + allData[i][allHeaders.indexOf('Timestamp')] + ']: ' +
          allData[i][allHeaders.indexOf('Body')]
        );
      }
    }

    const contextStr = threadContext.slice(-20).join('\n\n---\n\n'); // last 20 msgs

    const prompt = 'You are analyzing an email thread about engineering issue ' + issueId + '.\n\n' +
      'Email thread:\n' + contextStr + '\n\n' +
      'Extract intelligence and return ONLY raw JSON (no markdown):\n' +
      '{"decisions":["list of clear decisions made"],' +
      '"actionItems":[{"person":"name or email","action":"what they will do","deadline":"if mentioned"}],' +
      '"rootCause":"identified root cause or empty string",' +
      '"blockers":["list of blockers mentioned"],' +
      '"suggestedStatus":"Open|WIP|Resolved|Pending|or empty",' +
      '"sentiment":"Urgent|Neutral|Positive|Blocked|Resolved",' +
      '"newPeopleLooped":["email addresses mentioned or CC\'d"],' +
      '"keyPoints":["top 3 key discussion points"],' +
      '"summary":"2-3 sentence plain English summary of the full thread"}';

    const gemResult = callGemini(prompt);
    if (!gemResult.success) return;

    const ext = gemResult.result;
    const extSheet = ss.getSheetByName(EMAIL_SHEETS.EXTRACTIONS);

    // Find existing extraction row for this issue and update, or insert new
    const extData = extSheet.getDataRange().getValues();
    const extHeaders = extData[0];
    let existingRow = -1;
    for (let i = 1; i < extData.length; i++) {
      if (extData[i][extHeaders.indexOf('IssueID')] === issueId) { existingRow = i + 1; break; }
    }

    const rowData = [
      issueId,
      JSON.stringify(ext.decisions || []),
      JSON.stringify(ext.actionItems || []),
      ext.rootCause || '',
      JSON.stringify(ext.blockers || []),
      ext.suggestedStatus || '',
      ext.sentiment || 'Neutral',
      JSON.stringify(ext.newPeopleLooped || []),
      JSON.stringify(ext.keyPoints || []),
      ext.summary || '',
      new Date().toISOString()
    ];

    if (existingRow > 0) {
      extSheet.getRange(existingRow, 1, 1, rowData.length).setValues([rowData]);
    } else {
      extSheet.appendRow(rowData);
    }

    // Auto-apply suggested status if confidence is high
    if (ext.suggestedStatus && ext.sentiment === 'Resolved') {
      try { updateIssueStatus(issueId, 'Resolved', 'Auto-resolved based on email thread analysis by AI.'); } catch(e) {}
    }

  } catch (err) {
    console.error('extractEmailIntelligence error:', err.message);
  }
}

function getEmailSummary(issueId) {
  try {
    const ss = getSpreadsheet();
    const extSheet = ss.getSheetByName(EMAIL_SHEETS.EXTRACTIONS);
    if (!extSheet || extSheet.getLastRow() < 2) return { success: true, extraction: null };

            summary:         data[i][headers.indexOf('Summary')],

        };
      }

    return { success: true, extraction: null };
  } catch (err) {
    return { success: false, error: err.message };
  }
}

function getEmailParticipants(issueId) {
  try {
    const threadsResult = getLinkedThreads(issueId);
    if (!threadsResult.success) return { success: true, participants: [] };
    const all = [];
    threadsResult.threads.forEach(function(t) {
      (t.participants || []).forEach(function(p) {
        if (all.indexOf(p) === -1) all.push(p);
      });
    });
    return { success: true, participants: all };
  } catch (err) {
    return { success: false, error: err.message };
  }
}

// INBOUND EMAIL TRIAGE — Watches Gmail Label
// ============================================================

function watchInboundEmails() {
  try {
    if (getSetting('EMAIL_TRIAGE_ENABLED') !== 'true') return;

    const label = getSetting('EMAIL_TRIAGE_LABEL') || 'TechTrack-Inbox';
    let gmailLabel;
    try {
      gmailLabel = GmailApp.getUserLabelByName(label);
    } catch(e) {}

    if (!gmailLabel) {
      // Fall back to searching unread emails in inbox mentioning bugs/issues
      const threads = GmailApp.search('is:unread subject:(bug OR issue OR error OR crash OR broken OR failed) -label:techtrack-processed', 0, 10);
      threads.forEach(function(t) { triageInboundThread(t); });
      return;
    }

    const threads = gmailLabel.getThreads(0, 20);
    threads.forEach(function(t) { triageInboundThread(t); });

  } catch (err) {
    console.error('watchInboundEmails error:', err.message);
  }
}

function triageInboundThread(thread) {
  try {
    const messages = thread.getMessages();
    const first = messages[0];
    const subject = first.getSubject();
    const body = first.getPlainBody();
    const sender = first.getFrom();
    const senderEmail = extractEmail(sender);

    // Skip if already processed (has TECH- ID in subject)
    if (/\[TECH-\d+\]/.test(subject)) return;

    // Check rules first
    const rulesResult = getInboundRules();
    let ruleMatch = null;
    if (rulesResult.success) {
      ruleMatch = rulesResult.rules.find(function(r) {
        if (!r.active) return false;
        if (r.matchField === 'sender' && senderEmail.toLowerCase().indexOf(r.matchValue.toLowerCase()) !== -1) return true;
        if (r.matchField === 'subject' && subject.toLowerCase().indexOf(r.matchValue.toLowerCase()) !== -1) return true;
        if (r.matchField === 'domain') {
          const domain = senderEmail.split('@')[1] || '';
          return domain.toLowerCase() === r.matchValue.toLowerCase();
        }
        return false;
      });
    }

    const prompt = 'You are triaging an inbound email to create a TechTrack engineering issue.\n\n' +
      'From: ' + sender + '\nSubject: ' + subject + '\nBody:\n' + body.substring(0, 1500) + '\n\n' +
      'Return ONLY raw JSON:\n' +
      '{"shouldCreate":true,' +
      '"title":"concise issue title",' +
      '"description":"expanded clear description",' +
      '"priority":"Critical|High|Medium|Low",' +
      '"module":"API|Dashboard|Authentication|Database|UI|Backend|Frontend|DevOps|Other",' +
      '"environment":"Production|Staging|Development|Unknown",' +
      '"impact":"Revenue|Customer|Internal|Performance|Security|Unknown",' +
      '"isActualIssue":true,' +
      '"confidence":0.9,' +
      '"reason":"why you classified it this way"}';

    const gemResult = callGemini(prompt);
    if (!gemResult.success || !gemResult.result.shouldCreate || !gemResult.result.isActualIssue) {
      // Mark as processed anyway so we don't re-process
      try { thread.addLabel(GmailApp.createLabel('techtrack-processed')); } catch(e) {}
      return;
    }

    const ext = gemResult.result;

    // Apply rule overrides
    if (ruleMatch) {
      if (ruleMatch.priority) ext.priority = ruleMatch.priority;
      if (ruleMatch.module) ext.module = ruleMatch.module;
    }

    // Create the issue
    const issueId = generateIssueId();
    const ss = getSpreadsheet();
    const issuesSheet = ss.getSheetByName(SHEETS.ISSUES);
    const now = new Date();
    const slaHours = getSLAHours(ext.priority);

    issuesSheet.appendRow([
      issueId, ext.title, ext.description,
      ext.module, ext.priority, 'Open', ext.environment,
      senderEmail, ruleMatch ? ruleMatch.assignTo : '',
      now.toISOString(), '', '', '',
      ext.impact, slaHours, ''
    ]);

    logActivity(issueId, 'Created from inbound email triage', 'system');

    // Link the thread
    linkThreadToIssue(issueId, thread.getId(), subject, [senderEmail]);

    // Label as processed
    try { thread.addLabel(GmailApp.createLabel('techtrack-processed')); } catch(e) {}

    // Send confirmation to sender
    try {
      GmailApp.sendEmail(senderEmail,
        'Re: [' + issueId + '] ' + subject,
        'Your issue has been logged in TechTrack as ' + issueId + ': ' + ext.title + '. Our team will review it shortly.',
        { name: 'TechTrack Platform' }
      );
    } catch(e) {}

    console.log('Created issue ' + issueId + ' from inbound email: ' + subject);

  } catch (err) {
    console.error('triageInboundThread error:', err.message);
  }
}

function extractEmail(str) {
  const match = (str || '').match(/<([^>]+)>/);
  return match ? match[1] : (str || '').trim();
}

function extractName(str) {
  const match = (str || '').match(/^([^<]+)</);
  return match ? match[1].trim().replace(/^"|"$/g,'') : '';
}

function cleanEmailBody(body) {
  if (!body) return '';
  // Remove quoted text (lines starting with >)
  const lines = body.split('\n');
  const clean = [];
  let quoteBlock = false;
  for (let i = 0; i < lines.length; i++) {
    const line = lines[i];
    if (line.trim().startsWith('>')) { quoteBlock = true; continue; }
    if (quoteBlock && line.trim() === '') continue;
    quoteBlock = false;
    // Stop at common signature markers
    if (/^--\s*$/.test(line.trim()) || /^On .+ wrote:/.test(line)) break;
    clean.push(line);
  }
  return clean.join('\n').trim().substring(0, 2000);
}

function safeJsonParse(str, fallback) {
  try { return JSON.parse(str); } catch(e) { return fallback; }
}

function getWorkloadData() {
  try {
    const devsResult = getDevelopers();
    if (!devsResult.success) return devsResult;
    const issuesResult = getIssues(null);
    const issues = issuesResult.success ? issuesResult.issues : [];
    const open = issues.filter(i => !['Resolved','Closed'].includes(i.status));
    const workload = devsResult.developers.map(dev => {
      const assigned = open.filter(i => i.assignedTo === dev.email);
      const critical = assigned.filter(i => i.priority === 'Critical').length;
      const high = assigned.filter(i => i.priority === 'High').length;
      const breached = assigned.filter(i => i.slaBreached).length;
      const load = (critical * 4) + (high * 2) + (assigned.length - critical - high);
      return {
        email: dev.email, name: dev.name, team: dev.team,
        total: assigned.length, critical, high, breached,
        maxTickets: parseInt(dev.maxTickets) || 10,
        loadScore: load,
        issues: assigned.map(i => ({ id: i.id, title: i.title, priority: i.priority, status: i.status }))
      };
    });
    return { success: true, workload };
  } catch (err) { return { success: false, error: err.message }; }
}

function getMTTRStats() {
  try {
    const result = getIssues(null);
    if (!result.success) return result;
    const resolved = result.issues.filter(i => ['Resolved','Closed'].includes(i.status) && i.resolvedDate && i.createdDate);
    const byPriority = { Critical:[], High:[], Medium:[], Low:[] };
    const byModule = {};
    const byDev = {};
    resolved.forEach(i => {
      const hrs = (new Date(i.resolvedDate) - new Date(i.createdDate)) / 3600000;
      if (byPriority[i.priority]) byPriority[i.priority].push(hrs);
      if (!byModule[i.module]) byModule[i.module] = [];
      byModule[i.module].push(hrs);
      if (i.assignedTo) {
        if (!byDev[i.assignedTo]) byDev[i.assignedTo] = [];
        byDev[i.assignedTo].push(hrs);
      }
    });
    const avg = arr => arr.length ? Math.round(arr.reduce((a,b)=>a+b,0)/arr.length*10)/10 : 0;
    const priorityMTTR = {};
    Object.keys(byPriority).forEach(k => priorityMTTR[k] = avg(byPriority[k]));
    const moduleMTTR = {};
    Object.keys(byModule).forEach(k => moduleMTTR[k] = avg(byModule[k]));
    const devMTTR = {};
    Object.keys(byDev).forEach(k => devMTTR[k] = avg(byDev[k]));
    return { success: true, priorityMTTR, moduleMTTR, devMTTR, totalResolved: resolved.length };
  } catch (err) { return { success: false, error: err.message }; }
}

function getBurndownData(sprintId) {
  try {
    const result = getIssues(null);
    if (!result.success) return result;
    const issues = sprintId ? result.issues.filter(i => i.sprintId === sprintId) : result.issues;
    const now = new Date();
    const days = [];
    for (let d = 13; d >= 0; d--) {
      const date = new Date(now);
      date.setDate(date.getDate() - d);
      const dateStr = date.toISOString().split('T')[0];
      const remaining = issues.filter(i => {
        const created = new Date(i.createdDate).toISOString().split('T')[0];
        if (created > dateStr) return false;
        if (['Resolved','Closed'].includes(i.status) && i.resolvedDate) {
          return new Date(i.resolvedDate).toISOString().split('T')[0] > dateStr;
        }
        return true;
      }).length;
      days.push({ date: dateStr, remaining });
    }
    return { success: true, burndown: days };
  } catch (err) { return { success: false, error: err.message }; }
}

function runEscalationCheck() {
  try {
    const rulesResult = getEscalationRules();
    if (!rulesResult.success || !rulesResult.rules.length) return;
    const issuesResult = getIssues(null);
    if (!issuesResult.success) return;
    const now = new Date();
    const openIssues = issuesResult.issues.filter(i => !['Resolved','Closed'].includes(i.status));
    rulesResult.rules.forEach(rule => {
      if (!rule.active) return;
      openIssues.forEach(issue => {
        if (issue.priority !== rule.priority) return;
        const created = new Date(issue.createdDate);
        const hoursOpen = (now - created) / 3600000;
        if (hoursOpen < rule.hoursToEscalate) return;
        // Check if already escalated recently (last 2h)
        const logResult = getActivityLog(issue.id);
        const logs = logResult.success ? logResult.logs : [];
        const recentEsc = logs.find(l => l.action.includes('Escalated') &&
          (now - new Date(l.timestamp)) < 7200000);
        if (recentEsc) return;
        // Escalate
        const msg = 'Issue ' + issue.id + ' (' + issue.priority + ') has been open for ' +
          Math.round(hoursOpen) + ' hours without resolution: ' + issue.title;
        if (rule.escalateTo) {
          try {
            GmailApp.sendEmail(rule.escalateTo, '[TechTrack] ESCALATION: ' + issue.id, msg,
              { name: 'TechTrack Escalation' });
          } catch(e) {}
        }
        sendSlackAlert('🚨 *ESCALATION*\n*Issue:* ' + issue.id + '\n*Priority:* ' + issue.priority +
          '\n*Open for:* ' + Math.round(hoursOpen) + 'h\n*Title:* ' + issue.title);
        logActivity(issue.id, 'Escalated after ' + Math.round(hoursOpen) + 'h — sent to ' + (rule.escalateTo||'Slack'), 'system');
        if (rule.action === 'bump-priority') {
          const priorities = ['Low','Medium','High','Critical'];
          const idx = priorities.indexOf(issue.priority);
          if (idx < priorities.length - 1) {
            updateIssuePriority(issue.id, priorities[idx+1]);
          }
        }
      });
    });
  } catch (err) { console.error('runEscalationCheck error:', err.message); }
}

      if (data[i][h.indexOf('TemplateID')] === templateId) { sheet.deleteRow(i+1); return { success: true }; }
    }
    return { success: false, error: 'Not found' };
  } catch (err) { return { success: false, error: err.message }; }
}

function calculateNextRun(schedule) {
  const now = new Date();
  if (schedule === 'daily')   { now.setDate(now.getDate()+1); }
  else if (schedule === 'weekly')  { now.setDate(now.getDate()+7); }
  else if (schedule === 'monthly') { now.setMonth(now.getMonth()+1); }
  else if (schedule === 'weekday') {
    now.setDate(now.getDate()+1);
    while (now.getDay()===0||now.getDay()===6) now.setDate(now.getDate()+1);
  }
  return now.toISOString();
}

function runRecurringTemplates() {
  try {
    const result = getRecurringTemplates();
    if (!result.success) return;
    const now = new Date();
    const ss = getSpreadsheet();
    const sheet = ss.getSheetByName('RECURRING_TEMPLATES');
    const data = sheet.getDataRange().getValues();
    const h = data[0];
    result.templates.forEach((tpl, idx) => {
      if (!tpl.active) return;
      if (!tpl.nextRun || new Date(tpl.nextRun) > now) return;
      // Create the issue
      const issueId = generateIssueId();
      const issuesSheet = ss.getSheetByName(SHEETS.ISSUES);
      const slaHours = getSLAHours(tpl.priority);
      issuesSheet.appendRow([issueId, tpl.title, tpl.description, tpl.module,
        tpl.priority, 'Open', '', 'system', tpl.assignTo||'',
        now.toISOString(), '','','', '', slaHours, '']);
      logActivity(issueId, 'Auto-created from recurring template: ' + tpl.id, 'system');
      // Update nextRun
      const nextRun = calculateNextRun(tpl.schedule);
      for (let i = 1; i < data.length; i++) {
        if (data[i][h.indexOf('TemplateID')] === tpl.id) {
          sheet.getRange(i+1, h.indexOf('NextRun')+1).setValue(nextRun);
          break;
        }
      }
      console.log('Auto-created issue ' + issueId + ' from template ' + tpl.id);
    });
  } catch (err) { console.error('runRecurringTemplates error:', err.message); }
}

function syncIssuesToCalendar() {
  try {
    const calendarId = getSetting('GOOGLE_CALENDAR_ID');
    if (!calendarId || calendarId === 'false') return { success: false, error: 'Calendar ID not set' };
    const calendar = CalendarApp.getCalendarById(calendarId);
    if (!calendar) return { success: false, error: 'Calendar not found' };
    const result = getIssues(null);
    if (!result.success) return result;
    const openIssues = result.issues.filter(i => !['Resolved','Closed'].includes(i.status) && i.slaDeadline);
    let synced = 0;
    openIssues.forEach(issue => {
      try {
        const deadline = new Date(issue.slaDeadline);
        const title = '[SLA] ' + issue.id + ': ' + issue.title.substring(0,50);
        // Check if event already exists
        const existing = calendar.getEventsForDay(deadline, { search: issue.id });
        if (existing.length === 0) {
          calendar.createAllDayEvent(title, deadline, {
            description: 'SLA deadline for TechTrack issue ' + issue.id + '\nPriority: ' + issue.priority + '\n' + issue.title,
            color: issue.priority === 'Critical' ? CalendarApp.EventColor.RED :
                   issue.priority === 'High' ? CalendarApp.EventColor.ORANGE :
                   CalendarApp.EventColor.BLUE
          });
          synced++;
        }
      } catch(e) {}
    });
    return { success: true, synced };
  } catch (err) { return { success: false, error: err.message }; }
}

function linkCommitToIssue(issueId, commitData) {
  try {
    const ss = getSpreadsheet();
    const sheet = ss.getSheetByName('GIT_COMMITS');
    const linkId = 'GIT-' + Utilities.getUuid().substring(0,6).toUpperCase();
    sheet.appendRow([linkId, issueId, commitData.sha||'', commitData.message||'',
      commitData.author||'', commitData.repo||'', commitData.url||'', new Date().toISOString()]);
    logActivity(issueId, 'Commit linked: ' + (commitData.sha||'').substring(0,7) + ' — ' + (commitData.message||'').substring(0,60), 'system');
    // Auto-resolve if commit message contains "fixes #TECH-XXXX" or "closes #TECH-XXXX"
    const msg = (commitData.message||'').toLowerCase();
    if (/\b(fixes|closes|resolves)\b/.test(msg)) {
      updateIssueStatus(issueId, 'Resolved', 'Auto-resolved via git commit: ' + (commitData.sha||'').substring(0,7));
    }
    return { success: true, linkId };
  } catch (err) { return { success: false, error: err.message }; }
}

function getIssueCommits(issueId) {
  try {
    const ss = getSpreadsheet();

    if (!sheet || sheet.getLastRow() < 2) return { success: true, commits: [] };
    const data = sheet.getDataRange().getValues();
    const h = data[0];
    const commits = [];
    for (let i = 1; i < data.length; i++) {
      if (data[i][h.indexOf('IssueID')] === issueId) {

          message:   data[i][h.indexOf('Message')],
          author:    data[i][h.indexOf('Author')],
          repo:      data[i][h.indexOf('Repo')],
          url:       data[i][h.indexOf('URL')],
          linkedAt:  data[i][h.indexOf('LinkedAt')]
        });
      }
    }
    return { success: true, commits };
  } catch (err) { return { success: false, error: err.message }; }
}

// GitHub webhook handler — call via doPost
function doPost(e) {
  try {
    const payload = JSON.parse(e.postData.contents);
    // Handle GitHub push event
    if (payload.commits) {
      payload.commits.forEach(commit => {
        const msg = commit.message || '';
        // Find TECH- references
        const matches = msg.match(/TECH-\d{4}/gi) || [];
        matches.forEach(issueId => {
          linkCommitToIssue(issueId.toUpperCase(), {
            sha: commit.id,
            message: msg,
            author: (commit.author||{}).name || commit.author,
            repo: (payload.repository||{}).full_name || '',
            url: commit.url || ''
          });
        });
      });
    }
    return ContentService.createTextOutput('OK');
  } catch (err) {
    return ContentService.createTextOutput('Error: ' + err.message);
  }
}

// SMART PRIORITY AUTO-ESCALATION (AI)
// ============================================================

function runSmartEscalation() {
  try {
    const key = PropertiesService.getScriptProperties().getProperty('GEMINI_API_KEY');
    if (!key) return;
    if (getSetting('SMART_ESCALATION_ENABLED') !== 'true') return;
    const result = getIssues(null);
    if (!result.success) return;
    const stale = result.issues.filter(i => {
      if (['Resolved','Closed'].includes(i.status)) return false;
      const hrs = (new Date() - new Date(i.createdDate)) / 3600000;
      const slaHrs = i.slaHours || 24;
      return hrs > slaHrs * 0.8; // 80%+ of SLA used
    }).slice(0, 20);
    if (!stale.length) return;
    const list = stale.map(i =>
      i.id+'|'+i.title+'|'+i.priority+'|'+i.status+'|'+Math.round((new Date()-new Date(i.createdDate))/3600000)+'h open'
    ).join('\n');
    const prompt = 'These engineering issues are approaching or past their SLA. Identify which ones need priority escalation.\n\n' +
      list + '\n\n' +
      'Return ONLY raw JSON:\n' +
      '{"escalate":[{"id":"TECH-XXXX","from":"current priority","to":"new priority","reason":"why"}],' +
      '"noAction":["list of IDs that are fine as-is"]}';
    const r = callGemini(prompt);
    if (!r.success || !r.result.escalate) return;
    r.result.escalate.forEach(item => {
      if (item.to && item.to !== item.from) {
        updateIssuePriority(item.id, item.to);
        logActivity(item.id, 'AI auto-escalated priority: ' + item.from + ' → ' + item.to + '. Reason: ' + (item.reason||''), 'system');
      }
    });
  } catch (err) { console.error('runSmartEscalation error:', err.message); }
}

// ============================================================

function getSprintVelocity() {
  try {
    const sprintsResult = getSprints();
    if (!sprintsResult.success) return sprintsResult;
    const issuesResult = getIssues(null);
    if (!issuesResult.success) return issuesResult;
    const velocity = sprintsResult.sprints.map(sprint => {
      const sprintIssues = issuesResult.issues.filter(i => i.sprintId === sprint.id);
      const resolved = sprintIssues.filter(i => ['Resolved','Closed'].includes(i.status));
      const total = sprintIssues.length;
      const completionRate = total > 0 ? Math.round((resolved.length / total) * 100) : 0;
      const avgResHrs = resolved.filter(i => i.resolvedDate && i.createdDate).reduce((acc, i) => {
        return acc + (new Date(i.resolvedDate) - new Date(i.createdDate)) / 3600000;
      }, 0) / (resolved.length || 1);
      return {
        sprintId: sprint.id, sprintName: sprint.name,
        total, resolved: resolved.length, completionRate,
        avgResolutionHours: Math.round(avgResHrs)
      };
    });
    return { success: true, velocity };
  } catch (err) { return { success: false, error: err.message }; }
}

function getTimeLogs(issueId) {
  try {
    const ss = getSpreadsheet();

    if (!sheet || sheet.getLastRow() < 2) return { success: true, logs: [], totalHours: 0 };
    const data = sheet.getDataRange().getValues();
    const h = data[0];
    const logs = [];
    let total = 0;
    for (let i = 1; i < data.length; i++) {
      if (!issueId || data[i][h.indexOf('IssueID')] === issueId) {

        logs.push({

// ============================================================

function fullTextSearch(query) {
  try {
    if (!query || query.length < 2) return { success: true, results: [] };
    const q = query.toLowerCase();
    const issuesResult = getIssues(null);
    const commentsResult = getComments(null);
    const results = [];

    if (issuesResult.success) {
      issuesResult.issues.forEach(function(i) {
        var score = 0;
        var title = (i.title||'').toLowerCase();
        var desc  = (i.description||'').toLowerCase();
        var id    = (i.id||'').toLowerCase();
        if (id === q) score += 100;
        if (title.includes(q)) score += 50;
        if (desc.includes(q)) score += 20;
        q.split(' ').forEach(function(w) {
          if (w.length > 2 && title.includes(w)) score += 10;
          if (w.length > 2 && desc.includes(w)) score += 5;
        });
        if (score > 0) results.push({ type: 'issue', score, id: i.id, title: i.title, priority: i.priority, status: i.status, module: i.module });
      });
    }

    return {
      success: true,
      results: results.sort(function(a,b){ return b.score - a.score; }).slice(0, 20)
    };
  } catch(err) { return { success: false, error: err.message }; }
}

// MODULE HEALTH
// ============================================================
function getModuleHealthScores() {
  try {
    var result = getIssues(null); if (!result.success) return result;
    var mods = {};
    result.issues.forEach(function(i) {
      if (!i.module) return;
      if (!mods[i.module]) mods[i.module]={open:0,critical:0,breached:0,resolved:0,total:0};
      var m=mods[i.module]; m.total++;
      if (!['Resolved','Closed'].includes(i.status)) { m.open++; if(i.priority==='Critical')m.critical++; if(i.slaBreached)m.breached++; }
      else m.resolved++;
    });
    var scores = Object.keys(mods).map(function(name) {
      var m=mods[name];
      var score=100;
      if(m.total>0){score-=Math.round((m.open/m.total)*30);score-=m.critical*10;score-=m.breached*15;}
      score=Math.max(0,Math.min(100,score));
      var grade=score>=90?'A':score>=75?'B':score>=60?'C':score>=45?'D':'F';
      return {module:name,score,grade,open:m.open,critical:m.critical,breached:m.breached,resolved:m.resolved,total:m.total};
    });
    scores.sort(function(a,b){return a.score-b.score;});
    return {success:true,scores};
  } catch(err) { return {success:false,error:err.message}; }
}

// ============================================================
function getSLAComplianceReport(fromDate, toDate) {
  try {
    var result = getIssues(null); if (!result.success) return result;
    var from = fromDate?new Date(fromDate):new Date(Date.now()-30*24*3600000);
    var to   = toDate?new Date(toDate):new Date();
    var inRange = result.issues.filter(function(i){var d=new Date(i.createdDate);return d>=from&&d<=to;});
    var breached=inRange.filter(function(i){return i.slaBreached;}).length;
    var total=inRange.length; var withinSLA=total-breached;
    var pct=total>0?Math.round((withinSLA/total)*100):100;
    var byPriority={};
    ['Critical','High','Medium','Low'].forEach(function(p){
      var pi=inRange.filter(function(i){return i.priority===p;});
      var pb=pi.filter(function(i){return i.slaBreached;}).length;
      byPriority[p]={total:pi.length,breached:pb,compliance:pi.length>0?Math.round(((pi.length-pb)/pi.length)*100):100};
    });
    return {success:true,total,withinSLA,breached,compliancePct:pct,byPriority};
  } catch(err) { return {success:false,error:err.message}; }
}

function checkDueDateReminders() {
  try {
    var ss = getSpreadsheet();
    var sh = ss.getSheetByName('DUE_DATES');
    if (!sh || sh.getLastRow() < 2) return;
    var data = sh.getDataRange().getValues(), h = data[0];
    var now = new Date(), tomorrow = new Date(now.getTime() + 24*3600000);
    for (var i = 1; i < data.length; i++) {
      var due = new Date(data[i][h.indexOf('DueDate')]);
      var sent = data[i][h.indexOf('ReminderSent')];
      if (sent === true || sent === 'TRUE') continue;
      if (due <= tomorrow && due >= now) {
        var issueId = data[i][h.indexOf('IssueID')];
        var issueR = getIssueById(issueId);
        if (issueR.success && !['Resolved','Closed'].includes(issueR.issue.status)) {
          var email = issueR.issue.assignedTo || issueR.issue.raisedBy;
          if (email) {
            try {
              GmailApp.sendEmail(email, '[TechTrack] Due tomorrow: ' + issueId,
                issueId + ' "' + issueR.issue.title + '" is due on ' + due.toLocaleDateString(),
                { name: 'TechTrack Platform' });
            } catch(e2) {}
          }
          sh.getRange(i+1, h.indexOf('ReminderSent')+1).setValue(true);

// ─── ROUND-ROBIN ASSIGNMENT ───────────────────────────────────
function roundRobinAssign(module_) {
  try {
    var ss = getSpreadsheet();
    var devsR = getDevelopers();
    if (!devsR.success || !devsR.developers.length) return { success: false, error: 'No developers' };
    var devs = devsR.developers;
    // Get last assigned index
    var idx = parseInt(getSetting('ROUND_ROBIN_INDEX') || '0');
    // Skip devs over their max tickets
    var openIssues = getIssues(null);
    var devCounts = {};
    if (openIssues.success) {
      openIssues.issues.forEach(function(i) { if (i.assignedTo) devCounts[i.assignedTo] = (devCounts[i.assignedTo]||0) + 1; });
    }
    var attempts = 0;
    while (attempts < devs.length) {
      var dev = devs[idx % devs.length];
      var max = parseInt(dev.maxTickets || '10');
      var cur = devCounts[dev.email] || 0;
      if (cur < max) {
        updateSetting('ROUND_ROBIN_INDEX', String((idx + 1) % devs.length));
        return { success: true, email: dev.email, name: dev.name };
      }
      idx++; attempts++;
    }
    // All at capacity — assign to least loaded
    var least = devs.reduce(function(a, b) { return (devCounts[a.email]||0) <= (devCounts[b.email]||0) ? a : b; });
    return { success: true, email: least.email, name: least.name };
  } catch(e) { return { success: false, error: e.message }; }
}

// ─── FIRST RESPONSE TIME ─────────────────────────────────────
function getFirstResponseStats() {
  try {
    var issuesR = getIssues(null); if (!issuesR.success) return issuesR;
    var commentsSheet = getSpreadsheet().getSheetByName(SHEETS.COMMENTS);
    if (!commentsSheet) return { success: true, avgHours: 0, byDev: {} };
    var cData = commentsSheet.getDataRange().getValues(), ch = cData[0];
    var firstComments = {};
    for (var i = 1; i < cData.length; i++) {
      var iid = cData[i][ch.indexOf('IssueID')];
      var ts  = new Date(cData[i][ch.indexOf('Timestamp')]);
      if (!firstComments[iid] || ts < firstComments[iid]) firstComments[iid] = ts;
    }
    var stats = [], byDev = {};
    issuesR.issues.forEach(function(issue) {
      var created = new Date(issue.createdDate);
      var firstReply = firstComments[issue.id];
      if (!firstReply) return;
      var hrs = (firstReply - created) / 3600000;
      if (hrs < 0) return;
      stats.push(hrs);
      var dev = issue.assignedTo || 'Unassigned';
      if (!byDev[dev]) byDev[dev] = [];
      byDev[dev].push(hrs);
    });
    var avg = stats.length ? Math.round(stats.reduce(function(a,b){ return a+b; },0) / stats.length * 10) / 10 : 0;
    var byDevAvg = {};
    Object.keys(byDev).forEach(function(d) {
      byDevAvg[d] = Math.round(byDev[d].reduce(function(a,b){ return a+b; },0) / byDev[d].length * 10) / 10;
    });
    return { success: true, avgHours: avg, byDev: byDevAvg, sampleSize: stats.length };
  } catch(e) { return { success: false, error: e.message }; }
}

// ─── REPEAT OFFENDERS ────────────────────────────────────────
function getRepeatOffenders() {
  try {
    var actSheet = getSpreadsheet().getSheetByName(SHEETS.ACTIVITY_LOG);
    if (!actSheet || actSheet.getLastRow() < 2) return { success: true, offenders: [] };
    var data = actSheet.getDataRange().getValues(), h = data[0], reopenCounts = {};
    for (var i = 1; i < data.length; i++) {
      var action = data[i][h.indexOf('Action')] || '';
      if (action.toLowerCase().includes('reopen')) {
        var id = data[i][h.indexOf('IssueID')];
        reopenCounts[id] = (reopenCounts[id]||0) + 1;
      }
    }
    var issuesR = getIssues(null); if (!issuesR.success) return issuesR;
    var offenders = issuesR.issues
      .filter(function(i) { return reopenCounts[i.id] >= 2; })
      .map(function(i) { return Object.assign({}, i, { reopenCount: reopenCounts[i.id] }); })
      .sort(function(a,b) { return b.reopenCount - a.reopenCount; });
    return { success: true, offenders: offenders };
  } catch(e) { return { success: false, error: e.message }; }
}

// ─── DEV LEADERBOARD ─────────────────────────────────────────
function getDevLeaderboard() {

    var board = devsR.developers.map(function(dev) {
      var myResolved = resolved.filter(function(i) { return i.assignedTo === dev.email; });

        .reduce(function(acc,i,_,arr) { return acc + (new Date(i.resolvedDate)-new Date(i.createdDate))/3600000/arr.length; }, 0);

      return { email: dev.email, name: dev.name||dev.email, resolved: myResolved.length, avgHours: Math.round(avgHrs*10)/10, reopens: reopens };
    }).sort(function(a,b) { return b.resolved - a.resolved; });
    return { success: true, board: board, period: startOfMonth.toLocaleDateString() };

// ─── CFD + BURNUP DATA ────────────────────────────────────────
function getCFDData() {
  try {
    var issuesR = getIssues(null); if (!issuesR.success) return issuesR;
    var days = 30, now = new Date();
    var statuses = ['Open','WIP','Testing','Resolved','Closed'];
    var daily = [];
    for (var d = days-1; d >= 0; d--) {
      var day = new Date(now); day.setDate(day.getDate()-d);
      var dayStr = day.toISOString().split('T')[0];
      var counts = {};
      statuses.forEach(function(s) { counts[s] = 0; });
      issuesR.issues.forEach(function(i) {
        var created = new Date(i.createdDate);
        if (created > day) return;
        var status = i.status;
        if (counts.hasOwnProperty(status)) counts[status]++;
        else counts['Open']++;
      });
      daily.push({ date: dayStr, counts: counts });
    }
    return { success: true, daily: daily, statuses: statuses };
  } catch(e) { return { success: false, error: e.message }; }
}

function upsertTicketToSheet(sheet, data){

 sheet.getDataRange().getValues();

 let rowIndex = -1;

 for(let i=1;i<rows.length;i++){

  if(
   String(rows[i][0]) ==
   String(data.ticketId)
  ){

   rowIndex = i+1;
   break;

  }

 }

 const issueData =
 buildIssueData(data);

 if(rowIndex == -1){

  const result =
  createIssue(issueData);

  sheet.appendRow([

   data.ticketId,
   data.title,
   data.client,
   data.csm,
   data.techOwner,
   data.priority,
   data.lastUpdated,
   result.issueId

  ]);

 }else{

  // Batch write all columns at once — one API call
  sheet.getRange(rowIndex, 2, 1, 6).setValues([[
   data.title,
   data.client,
   data.csm,
   data.techOwner,
   data.priority,
   data.lastUpdated
  ]]);

 }

function sendAIStandupDigest() {

  var issuesResult = getIssues(null);
  var usersResult = getUsers(true);
  if (!issuesResult.success || !usersResult.success) return { success: false, error: 'Could not load data' };

  var developers = usersResult.users.filter(function(u) { return u.role === 'Developer' || u.role === 'Admin'; });
  var allIssues = issuesResult.issues;
  var yesterday = new Date(Date.now() - 24 * 3600 * 1000);
  var sent = 0;

  developers.forEach(function(dev) {
    var myIssues = allIssues.filter(function(i) { return i.assignedTo === dev.email; });
    var openIssues = myIssues.filter(function(i) { return !['Resolved','Closed'].includes(i.status); });
    var resolvedYesterday = myIssues.filter(function(i) {
      return ['Resolved','Closed'].includes(i.status) && i.resolvedDate && new Date(i.resolvedDate) >= yesterday;
    });
    var slaAtRisk = openIssues.filter(function(i) { return i.slaRisk || i.slaBreached; });

    var prompt = 'Write a concise daily standup summary for an engineer.\n\n' +
      'Engineer: ' + dev.name + '\nOpen tickets: ' + openIssues.length + ' (' + slaAtRisk.length + ' at SLA risk)\n' +
      'Resolved yesterday: ' + resolvedYesterday.length + '\n\nOpen issues:\n' +
      openIssues.slice(0, 10).map(function(i) {
        return '- ' + i.id + ': ' + i.title + ' [' + i.priority + '][' + i.status + ']' + (i.slaBreached ? ' ⚠️ SLA BREACHED' : '');
      }).join('\n') + '\n\n' +
      'Return ONLY raw JSON:\n' +
      '{"greeting":"personalised morning greeting","focusToday":"top 2-3 things to focus on","slaWarning":"SLA summary if any else null","accomplishment":"what they resolved if any else null","tip":"one productivity tip"}';

    try {
      var r = callGemini(prompt);
      if (!r.success) return;
      var d;
      try { d = typeof r.result === 'string' ? JSON.parse(r.result.replace(/```json|```/g,'').trim()) : r.result; } catch(e) { d = {}; }

      var html = buildStandupEmailHtml(dev, d, openIssues, slaAtRisk, resolvedYesterday);

      });

    }
  });

}

    if (action === 'status') {
      sheet.getRange(row, headers.indexOf('Status') + 1).setValue(value);
      if (value === 'Resolved') sheet.getRange(row, headers.indexOf('ResolvedDate') + 1).setValue(new Date().toISOString());
      logActivity(id, 'Bulk status change to ' + value, currentUser.user.email);
    } else if (action === 'assign') {
      sheet.getRange(row, headers.indexOf('AssignedTo') + 1).setValue(value);
      logActivity(id, 'Bulk assigned to ' + value, currentUser.user.email);
    } else if (action === 'priority') {
      sheet.getRange(row, headers.indexOf('Priority') + 1).setValue(value);
      logActivity(id, 'Bulk priority set to ' + value, currentUser.user.email);
    }
    updated++;
  }

  return { success: true, updated: updated };
}

// ─── Feature 6: Developer Workload Board ─────────────────────
function getDevWorkloadBoard() {
  if (!isFeatureEnabled('WORKLOAD_BOARD_ENABLED')) {
    return { success: false, error: 'Workload board not enabled. Enable WORKLOAD_BOARD_ENABLED in Admin > Settings.' };
  }
  var devsResult = getUsers(true);
  if (!devsResult.success) return devsResult;
  var devs = devsResult.users.filter(function(u) { return u.role === 'Developer' || u.role === 'Admin'; });

  var issuesResult = getIssues(null);
  var allIssues = issuesResult.success ? issuesResult.issues : [];

  var board = devs.map(function(dev) {
    var myIssues = allIssues.filter(function(i) {
      return i.assignedTo === dev.email && !['Resolved','Closed'].includes(i.status);
    });
    var resolved = allIssues.filter(function(i) {
      return i.assignedTo === dev.email && ['Resolved','Closed'].includes(i.status);
    });
    var slaAtRisk = myIssues.filter(function(i) { return i.slaBreached || i.slaRisk; });
    var critical = myIssues.filter(function(i) { return i.priority === 'Critical'; });

    var maxTickets = parseInt(dev.maxTickets) || 10;
    var capacityPct = Math.min(100, Math.round((myIssues.length / maxTickets) * 100));

    var avgMTTR = null;
    var hrs = resolved
      .filter(function(i) { return i.resolvedDate && i.createdDate; })
      .map(function(i) { return (new Date(i.resolvedDate) - new Date(i.createdDate)) / 3600000; });
    if (hrs.length > 0) avgMTTR = Math.round(hrs.reduce(function(a,b){return a+b;},0) / hrs.length);

    return {
      email: dev.email,
      name: dev.name,
      team: dev.team || '',
      skill: dev.skill || '',
      openCount: myIssues.length,
      maxTickets: maxTickets,
      capacityPct: capacityPct,
      slaAtRisk: slaAtRisk.length,
      critical: critical.length,
      resolvedTotal: resolved.length,
      avgMTTRHours: avgMTTR,
      issues: myIssues.slice(0, 5)
    };
  });

  board.sort(function(a, b) { return b.capacityPct - a.capacityPct; });
  return { success: true, board: board };
}

// ─── Feature 14: SLA Breach Predictor ────────────────────────
function predictSLABreaches() {
  if (!isFeatureEnabled('SLA_PREDICTOR_ENABLED')) {
    return { success: false, error: 'SLA predictor not enabled. Enable SLA_PREDICTOR_ENABLED in Admin > Settings.' };
  }
  var issuesResult = getIssues(null);
  if (!issuesResult.success) return issuesResult;

  var mttrResult = getMTTRStats ? getMTTRStats() : { success: false };
  var priorityMTTR = (mttrResult.success && mttrResult.priorityMTTR) ? mttrResult.priorityMTTR : {};

  var openIssues = issuesResult.issues.filter(function(i) {
    return !['Resolved','Closed','Need Info','Blocked'].includes(i.status);
  });

  var predictions = openIssues.map(function(issue) {
    var expectedMTTR = priorityMTTR[issue.priority] || issue.slaHours || 24;
    var ageHours = (Date.now() - new Date(issue.createdDate)) / 3600000;
    var predictedResolutionInHours = Math.max(0, expectedMTTR - ageHours);
    var hoursRemaining = issue.slaHoursRemaining || 0;
    var willBreachSLA = predictedResolutionInHours > hoursRemaining && !issue.slaBreached;
    var riskScore = hoursRemaining > 0 ? Math.min(100, Math.round((predictedResolutionInHours / hoursRemaining) * 100)) : 100;

    return {
      issueId: issue.id,
      title: issue.title,
      priority: issue.priority,
      assignedTo: issue.assignedTo,
      slaHoursRemaining: hoursRemaining,
      predictedResolutionHours: Math.round(predictedResolutionInHours),
      willBreachSLA: willBreachSLA,
      riskScore: riskScore
    };
  }).filter(function(p) { return p.willBreachSLA || p.riskScore > 80; });

  predictions.sort(function(a, b) { return b.riskScore - a.riskScore; });
  return { success: true, predictions: predictions };
}

// ─── Batch Update Issues ─────────────────────────────────────
function batchUpdateIssues(issueIds, updates) {
  try {
    var ss = getSpreadsheet();
    var sheet = ss.getSheetByName(SHEETS.ISSUES);
    var data = sheet.getDataRange().getValues();
    var headers = data[0];
    var currentUser = getCurrentUser();
    if (!currentUser.success) return currentUser;

    var allowedFields = { status: 'Status', priority: 'Priority', assignedTo: 'AssignedTo', module: 'Module' };
    var updated = 0;

    for (var i = 1; i < data.length; i++) {
      var rowId = data[i][headers.indexOf('IssueID')];
      if (issueIds.indexOf(rowId) === -1) continue;
      var rowNum = i + 1;

      Object.keys(updates).forEach(function(field) {
        var col = allowedFields[field];
        if (!col) return;
        var colIdx = headers.indexOf(col);
        if (colIdx === -1) return;
        sheet.getRange(rowNum, colIdx + 1).setValue(updates[field]);
      });

      if (updates.status) {
        var now = new Date().toISOString();
        if (updates.status === 'WIP' && !data[i][headers.indexOf('StartedDate')]) {
          sheet.getRange(rowNum, headers.indexOf('StartedDate') + 1).setValue(now);
        }
        if (updates.status === 'Resolved') {
          sheet.getRange(rowNum, headers.indexOf('ResolvedDate') + 1).setValue(now);
        }
        if (updates.status === 'Closed') {
          sheet.getRange(rowNum, headers.indexOf('ClosedDate') + 1).setValue(now);
        }
      }

      var changes = Object.keys(updates).map(function(k) { return k + '=' + updates[k]; }).join(', ');
      logActivity(rowId, 'Batch update: ' + changes, currentUser.user.email);
      updated++;
    }

    return { success: true, updated: updated };
  } catch (err) {
    return { success: false, error: err.message };
  }
}

        time: row[headers.indexOf('CreatedDate')],
        read: row[headers.indexOf('IsRead')] === true || row[headers.indexOf('IsRead')] === 'TRUE'
      });
      if (notifications.length >= 50) break;
    }
    return { success: true, notifications: notifications };
  } catch (err) {
    return { success: false, error: err.message };
  }
}

// ─── MTTR Distribution ───────────────────────────────────────
function getMTTRDistribution() {
  try {
    var result = getIssues(null);
    if (!result.success) return result;
    var buckets = {
      '<4h': 0, '4-8h': 0, '8-24h': 0, '1-3d': 0, '3-7d': 0, '>7d': 0
    };
    var resolved = result.issues.filter(function(i) {
      return i.resolvedDate && i.createdDate;
    });
    resolved.forEach(function(i) {
      var hours = (new Date(i.resolvedDate) - new Date(i.createdDate)) / 3600000;
      if (hours < 4) buckets['<4h']++;
      else if (hours < 8) buckets['4-8h']++;
      else if (hours < 24) buckets['8-24h']++;
      else if (hours < 72) buckets['1-3d']++;
      else if (hours < 168) buckets['3-7d']++;
      else buckets['>7d']++;
    });
    return { success: true, distribution: buckets, totalResolved: resolved.length };
  } catch (err) {
    return { success: false, error: err.message };
  }
}

// ─── Module Risk Score ───────────────────────────────────────
function getModuleRiskScores() {
  try {
    var result = getIssues(null);
    if (!result.success) return result;
    var modules = {};
    var open = result.issues.filter(function(i) { return !['Resolved','Closed'].includes(i.status); });
    open.forEach(function(i) {
      var m = i.module || 'Unknown';
      if (!modules[m]) modules[m] = { critical: 0, high: 0, medium: 0, low: 0, slaBreached: 0, total: 0 };
      modules[m].total++;
      modules[m][i.priority ? i.priority.toLowerCase() : 'low']++;
      if (i.slaBreached) modules[m].slaBreached++;
    });
    var scores = Object.keys(modules).map(function(m) {
      var d = modules[m];
      var score = (d.critical * 40) + (d.high * 20) + (d.medium * 5) + (d.low * 1) + (d.slaBreached * 30);
      var risk = score > 100 ? 'Critical' : score > 50 ? 'High' : score > 20 ? 'Medium' : 'Low';
      return { module: m, score: score, risk: risk, total: d.total, critical: d.critical, high: d.high, slaBreached: d.slaBreached };
    });
    scores.sort(function(a,b) { return b.score - a.score; });
    return { success: true, scores: scores };
  } catch (err) {
    return { success: false, error: err.message };
  }
}

// ─── Auto-close Resolved Issues ──────────────────────────────
function autoCloseResolvedIssues() {
  try {
    var autoCloseDays = parseInt(getSetting('AUTO_CLOSE_DAYS') || '7');
    if (!autoCloseDays || autoCloseDays <= 0) return { success: true, message: 'Auto-close disabled' };

    var ss = getSpreadsheet();
    var sheet = ss.getSheetByName(SHEETS.ISSUES);
    var data = sheet.getDataRange().getValues();
    var headers = data[0];
    var now = new Date();
    var closed = 0;

    for (var i = 1; i < data.length; i++) {
      if (data[i][headers.indexOf('Status')] !== 'Resolved') continue;
      var resolvedDate = data[i][headers.indexOf('ResolvedDate')];
      if (!resolvedDate) continue;
      var daysSinceResolved = (now - new Date(resolvedDate)) / 86400000;
      if (daysSinceResolved >= autoCloseDays) {
        var rowNum = i + 1;
        var issueId = data[i][headers.indexOf('IssueID')];
        sheet.getRange(rowNum, headers.indexOf('Status') + 1).setValue('Closed');
        sheet.getRange(rowNum, headers.indexOf('ClosedDate') + 1).setValue(now.toISOString());
        logActivity(issueId, 'Auto-closed after ' + autoCloseDays + ' days in Resolved state', 'system');
        closed++;
      }
    }
    return { success: true, closed: closed };
  } catch (err) {
    return { success: false, error: err.message };
  }
}

// ─── Daily Digest Data ────────────────────────────────────────
function getDailyDigestData() {
  try {
    var result = getIssues(null);
    if (!result.success) return result;
    var now = new Date();
    var yesterday = new Date(now - 86400000);
    var issues = result.issues;
    var open = issues.filter(function(i) { return !['Resolved','Closed'].includes(i.status); });
    var newToday = issues.filter(function(i) { return new Date(i.createdDate) >= yesterday; });
    var resolvedToday = issues.filter(function(i) { return i.resolvedDate && new Date(i.resolvedDate) >= yesterday; });
    var slaBreached = issues.filter(function(i) { return i.slaBreached; });
    var critical = open.filter(function(i) { return i.priority === 'Critical'; });
    return {
      success: true,
      digest: {
        totalOpen: open.length,
        newToday: newToday.length,
        resolvedToday: resolvedToday.length,
        slaBreached: slaBreached.length,
        criticalOpen: critical.length,
        unassigned: open.filter(function(i) { return !i.assignedTo; }).length,
        topIssues: critical.slice(0, 5).map(function(i) { return { id: i.id, title: i.title, priority: i.priority, status: i.status }; })
      }
    };
  } catch (err) {
    return { success: false, error: err.message };
  }
}

// ─── Issue Search (Enhanced) ─────────────────────────────────
function searchIssues(query) {
  try {
    if (!query || query.trim().length < 2) return { success: true, issues: [] };
    var result = getIssues(null);
    if (!result.success) return result;
    var q = query.toLowerCase().trim();
    var matches = result.issues.filter(function(i) {
      return (i.id && i.id.toLowerCase().indexOf(q) !== -1) ||
        (i.title && i.title.toLowerCase().indexOf(q) !== -1) ||
        (i.description && i.description.toLowerCase().indexOf(q) !== -1) ||
        (i.module && i.module.toLowerCase().indexOf(q) !== -1) ||
        (i.raisedBy && i.raisedBy.toLowerCase().indexOf(q) !== -1) ||
        (i.assignedTo && i.assignedTo.toLowerCase().indexOf(q) !== -1) ||
        (i.impact && i.impact.toLowerCase().indexOf(q) !== -1);
    }).slice(0, 20);
    return { success: true, issues: matches, query: query };
  } catch (err) {
    return { success: false, error: err.message };
  }
}

// ─── Watchlist Notification Broadcast ────────────────────────
function notifyWatchers(issueId, title, action, actorEmail) {
  try {
    var watchersResult = getWatchers(issueId);
    if (!watchersResult.success) return;
    var watchers = watchersResult.watchers.filter(function(w) { return w !== actorEmail; });
    if (watchers.length === 0) return;
    watchers.forEach(function(email) {
      try {
        createNotification(email, 'Update: ' + issueId, title + ' — ' + action, 'info', issueId);
        var notifEnabled = getSetting('NOTIF_ON_STATUS') !== 'false';
        if (notifEnabled) {
          GmailApp.sendEmail(email, '[TechTrack] Update on ' + issueId,
            'Issue ' + issueId + ' (' + title + ') has been updated: ' + action + '\n\nYou are watching this issue.');
        }
      } catch(e) {
        console.error('Failed to notify watcher ' + email + ': ' + e.message);
      }
    });
  } catch (err) {
    console.error('notifyWatchers error:', err.message);
  }
}

// ─── Get Summary Statistics (for digest / dashboard) ─────────
function getSummaryStats() {
  try {
    var result = getIssues(null);
    if (!result.success) return result;
    var now = new Date();
    var weekAgo = new Date(now - 7 * 86400000);
    var issues = result.issues;
    var open = issues.filter(function(i) { return !['Resolved','Closed'].includes(i.status); });
    var priorities = ['Critical','High','Medium','Low'];
    var byPriority = {};
    priorities.forEach(function(p) {
      byPriority[p] = open.filter(function(i) { return i.priority === p; }).length;
    });
    var statuses = ['Open','Acknowledged','WIP','Testing','Blocked'];
    var byStatus = {};
    statuses.forEach(function(s) {
      byStatus[s] = open.filter(function(i) { return i.status === s; }).length;
    });
    return {
      success: true,
      stats: {
        totalOpen: open.length,
        slaBreached: issues.filter(function(i) { return i.slaBreached; }).length,
        resolvedThisWeek: issues.filter(function(i) { return i.resolvedDate && new Date(i.resolvedDate) >= weekAgo; }).length,
        newThisWeek: issues.filter(function(i) { return new Date(i.createdDate) >= weekAgo; }).length,
        unassigned: open.filter(function(i) { return !i.assignedTo; }).length,
        byPriority: byPriority,
        byStatus: byStatus
      }
    };
  } catch (err) {
    return { success: false, error: err.message };
  }
}
